# -------------------- Constants and Configuration --------------------
API_KEY_ENV_VAR = "GEMINI_API_KEY"
REQUESTS_PER_MINUTE = 15
TOKENS_PER_MINUTE = 1_000_000
EMOJI_THINKING = "🤔"
EMOJI_SUCCESS = "✅"
EMOJI_CLOCK = "⏳"

# -------------------- Logging Setup --------------------
logging.basicConfig(
    level=logging.INFO, format="%(asctime)s - %(levelname)s - %(message)s"
)


# -------------------- Helper Functions --------------------
def configure_api_key():
    """Configure API key for Google Generative AI."""
    try:
        genai.configure(api_key=os.environ[API_KEY_ENV_VAR])
        logging.info("API key configured successfully. 🔑")
    except KeyError:
        logging.error(
            f"API key not found in environment variables. Please set the '{API_KEY_ENV_VAR}' environment variable."
        )
        exit(1)


def extract_text_from_pdf(pdf_path):
    """Extracts text from a PDF file."""
    logging.info(f"Extracting text from PDF: {pdf_path}")
    with open(pdf_path, "rb") as pdf_file:
        pdf_reader = PyPDF2.PdfReader(pdf_file)
        return "".join(
            page.extract_text() for page in pdf_reader.pages if page.extract_text()
        )


# -------------------- Rate Limiting Decorator --------------------
@contextmanager
def rate_limited():
    """Context manager for rate limiting."""
    yield
    time.sleep(60 / REQUESTS_PER_MINUTE)


# -------------------- Error Handling Decorator --------------------
def handle_gemini_errors(func):
    """Decorator to handle common errors from Gemini API calls."""

    def wrapper(*args, **kwargs):
        with rate_limited(), Halo(
                text=f"Executing {func.__name__}...", spinner="dots"
        ) as spinner:
            try:
                result = func(*args, **kwargs)
                spinner.succeed(f"Operation successful! {EMOJI_SUCCESS}")
                return result
            except HTTPError as http_err:
                spinner.fail(f"HTTP error occurred: {http_err}")
                logging.error(f"HTTP error occurred: {http_err}")
            except Timeout as timeout_err:
                spinner.fail(f"Request timed out: {timeout_err}")
                logging.error(f"Request timed out: {timeout_err}")
            except RequestException as req_err:
                spinner.fail(f"Request error occurred: {req_err}")
                logging.error(f"Request error occurred: {req_err}")
            except RuntimeError as runtime_err:
                # Re-raise RuntimeError to be caught in tests
                raise runtime_err
            except Exception as e:
                spinner.fail(f"An error occurred: {e}")
                logging.error(f"An error occurred: {e}")
            raise

    return wrapper



# -------------------- Gemini API Services --------------------
@handle_gemini_errors
@sleep_and_retry
@limits(calls=REQUESTS_PER_MINUTE, period=60)
def generate_text_from_text_only_input(
        model_name="gemini-1.5-flash", prompt="Write a story about a magic backpack."
):
    """Generates text from a text-only prompt."""
    logging.info(f"Generating text from prompt: {prompt}")
    model = genai.GenerativeModel(model_name)
    response = model.generate_content(prompt)
    return response.text

@sleep_and_retry
@limits(calls=REQUESTS_PER_MINUTE, period=60)

        model_name="gemini-1.5-flash",
        image_path="media/organ.jpg",
        prompt="Tell me about this instrument",
):
    """Generates text from a prompt and an image."""
    logging.info(f"Generating text from prompt: {prompt} and image: {image_path}")

    image = PIL.Image.open(image_path)
    response = model.generate_content([prompt, image])
    return response.text


@handle_gemini_errors
@sleep_and_retry
@limits(calls=REQUESTS_PER_MINUTE, period=60)
def generate_text_stream(
        model_name="gemini-1.5-flash", prompt="Write a story about a magic backpack."
):
    """Generates text in a stream."""
    logging.info(f"Generating text stream from prompt: {prompt}")
    model = genai.GenerativeModel(model_name)
    response = model.generate_content(prompt, stream=True)
    return "\n".join(chunk.text for chunk in response)


@handle_gemini_errors
@sleep_and_retry
@limits(calls=REQUESTS_PER_MINUTE, period=60)
def interactive_chat(model_name="gemini-1.5-flash", initial_history=None):
    """Simulates an interactive chat."""
    initial_history = initial_history or [
        {"role": "user", "parts": "Hello"},
        {"role": "model", "parts": "Great to meet you. What would you like to know?"},
    ]
    logging.info(f"Starting interactive chat with model: {model_name}")
    model = genai.GenerativeModel(model_name)
    chat = model.start_chat(history=initial_history)
    while True:
        user_input = input("You: ")
        if user_input == "exit":
            break
        print(f"Thinking {EMOJI_THINKING}...", end="")
        response = chat.send_message(user_input)
        print("\nModel:", response.text)
    return "Chat ended."


@handle_gemini_errors
@sleep_and_retry
@limits(calls=REQUESTS_PER_MINUTE, period=60)
def configure_text_generation(
        model_name="gemini-1.5-flash",
        prompt="Tell me a story about a magic backpack.",
        candidate_count=1,
        stop_sequences=None,
        max_output_tokens=20,
        temperature=1.0,
):
    """Configures text generation parameters."""
    stop_sequences = stop_sequences or ["x"]
    logging.info(
        f"Generating text with specific configuration: model={model_name}, prompt={prompt}, "
        f"candidate_count={candidate_count}, stop_sequences={stop_sequences}, max_output_tokens={max_output_tokens}, "
        f"temperature={temperature}"
    )
    generation_config = genai.types.GenerationConfig(
        candidate_count=candidate_count,
        stop_sequences=stop_sequences,
        max_output_tokens=max_output_tokens,
        temperature=temperature,
    )
    model = genai.GenerativeModel(model_name)
    response = model.generate_content(prompt, generation_config=generation_config)
    return response.text

@handle_gemini_errors
@sleep_and_retry
@limits(calls=REQUESTS_PER_MINUTE, period=60)
def prompt_with_multiple_documents(
        model_name="gemini-1.5-flash",
        document_paths=None,
        prompt="Summarize the differences between the thesis statements for these documents.",
):
    """Prompts with multiple documents."""
    document_paths = document_paths or ["example-1.pdf", "example-2.pdf"]
    logging.info(f"Prompting with multiple documents: {document_paths}")
    documents = [extract_text_from_pdf(path) for path in document_paths]
    model = genai.GenerativeModel(model_name)
    response = model.generate_content([prompt] + documents)
    return response.text

@handle_gemini_errors
@sleep_and_retry
@limits(calls=REQUESTS_PER_MINUTE, period=60)
def upload_and_generate_content_from_video(
        model_name="gemini-1.5-pro",
        video_path="GreatRedSpot.mp4",
        prompt="Summarize this video. Then create a quiz with answer key based on the information in the video.",
):
    """Uploads a video and generates content based on it."""
    logging.info(f"Uploading video: {video_path} and generating content from it.")
    video_file = genai.upload_file(video_path)
    while video_file.state.name == "PROCESSING":
        time.sleep(10)
        video_file = genai.get_file(video_file.name)
    if video_file.state.name == "FAILED":
        raise ValueError(f"Video processing failed: {video_file.state.name}")
    model = genai.GenerativeModel(model_name)
    response = model.generate_content(
        [video_file, prompt], request_options={"timeout": 600}
    )
    return response.text

@handle_gemini_errors
@sleep_and_retry
@limits(calls=REQUESTS_PER_MINUTE, period=60)
def execute_code(
        model_name="gemini-1.5-pro",
        prompt="What is the sum of the first 50 prime numbers? Generate and run code for the calculation, and make sure you get all 50.",
):
    """Executes Python code using the Gemini API."""
    logging.info("Executing Python code with Gemini API.")
    model = genai.GenerativeModel(model_name, tools="code_execution")
    response = model.generate_content(prompt)

    if " copyrighted material" in response.text:
        raise RuntimeError("The model was reciting from copyrighted material.")

    return response.text
